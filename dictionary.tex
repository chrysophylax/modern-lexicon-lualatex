% Modern kernel mark classes (LaTeX 2022+). Unlike legacy e-TeX \newmarks
% classes, these are tracked correctly across BOTH columns of a twocolumn
% page: \FirstMark[page]/\LastMark[page] give the true first/last mark of
% the whole page, not just of the second column's output cycle.
\NewMarkClass{dictword}

% Continuation marker: 'dictcont' carries the current headword while an
% entry is "open" (set by \printentry, cleared by \entrydone). If a page
% begins with a non-empty dictcont mark inherited from the previous page
% (\TopMark[page]), that page starts INSIDE a long entry and the running
% head gains a "(word, cont.)" note -- the classic print-dictionary
% solution for entries spanning multiple pages.
\NewMarkClass{dictcont}
% Emptiness is decided by typesetting the mark into a box and measuring
% it: \TopMark is not expandable inside \edef, so token-level tests
% (\ifx / \detokenize) always see the unexpanded command and misfire.
\newsavebox\contnotebox
\newcommand{\contnote}{%
  \sbox\contnotebox{\TopMark[page]{dictcont}}%
  \ifdim\wd\contnotebox>0pt
    \ \ (\usebox\contnotebox, cont.)%
  \fi}

\fancyhead[LE,RO]{\small\bfseries\thepage}

% --- ENTRY MACRO ---
% Prints nothing (not even empty brackets) when the IPA cell is empty
\newcommand{\phonetic}[1]{%
  \if\relax\detokenize{#1}\relax\else{\textcolor[rgb]{0.3,0.3,0.3}{[#1]}}\fi}
\newcommand{\pos}[1]{{\em #1}}

% NOTE: indexing moved out of this macro -- the Lua engine emits
% \index{sortkey@word} itself so the index follows the custom collation.
% \interlinepenalty forbids column/page breaks INSIDE the entry paragraph
% (dictionary convention: an entry never splits); breaks remain allowed
% between entries. Grouped so the penalty does not leak out.
% The marks are inserted AFTER the headword, i.e. once the paragraph has
% truly started. \InsertMark is a bare \marks node and does NOT trigger
% horizontal mode, so placed before \textbf it would still land in the
% vertical list ahead of the entry's first line -- and since a mark node
% is non-discardable, the baselineskip glue after it becomes a legal
% breakpoint: when the entry moves to the next column/page its marks
% legally stay behind, skewing the running head by one entry/page. In
% the horizontal list the marks migrate out glued AFTER the entry's
% first line and can never be separated from it.
\newcommand{\printentry}[4]{%
  {\par\dictfontsize
   \interlinepenalty=10000
   \hangindent=1.5em\hangafter=1%
   \textbf{#1}%
   \InsertMark{dictword}{#1}% Update header marks
   \InsertMark{dictcont}{#1}% Entry is now "open" for the continuation note
   \ \phonetic{#2} \pos{#3} #4\par}%
}

% --- SENSE-LEVEL FALLBACK (large entries) ---
% For entries whose definition exceeds SENSE_SPLIT_THRESHOLD (see the Lua
% engine), the keep-whole policy would leave huge column gaps or overflow
% the page. Fallback: the headword plus sense 1 stay one unbreakable block
% (\printentry above), and every further numbered sense becomes its own
% unbreakable paragraph typeset by \printsense. Column/page breaks are
% thereby allowed BETWEEN senses but never inside one -- the standard
% print-dictionary compromise for long entries.
\newcommand{\printsense}[1]{%
  {\par\dictfontsize
   \interlinepenalty=10000
   \hangindent=1.5em\hangafter=1%
   #1\par}%
}

% Closes an entry. Clearing the dictcont mark records "page top is
% between entries" so the running head shows no continuation note.
% The Lua engine injects this INSIDE the entry's final paragraph (last
% macro argument) so the mark migrates out glued to the last line -- as
% a stand-alone vertical-mode token it could be pushed to the next page
% while the entry stays, leaving the entry spuriously "open".
\newcommand{\entrydone}{\InsertMark{dictcont}{}}

% --- EXAMPLE LINE MACRO ---
% Charis SIL has no glyph for U+25C8, so fetch it from DejaVu Sans,
% scaled down slightly to sit well against the 9.5pt Charis text.
\newfontfamily\exmarkerfont{DejaVu Sans}[Scale=0.75]
\newcommand{\exmarker}{{\exmarkerfont ◈ }}

% One example per line, kept on the 14pt baseline grid via \dictfontsize.
% Indented to the entry hang (1.5em); wrapped lines hang a little further.
% \nopagebreak glues the example to the material above (its entry or the
% previous example); \interlinepenalty keeps the example line itself whole.
\newcommand{\printexample}[2]{%
  {\par\nopagebreak\dictfontsize
   \interlinepenalty=10000
   \leftskip=1.5em
   \hangindent=1em\hangafter=1%
   \noindent\exmarker\,\textbf{#1} #2\par}%
}

% Breakable variant for LARGE entries: chaining \nopagebreak through a
% long example list would create one giant unsplittable block, so only
% the first example is glued to its entry; the rest use this macro and
% may break onto the next column/page (each line still kept whole).
\newcommand{\printexamplecont}[2]{%
  {\par\dictfontsize
   \interlinepenalty=10000
   \leftskip=1.5em
   \hangindent=1em\hangafter=1%
   \noindent\exmarker\,\textbf{#1} #2\par}%
}

% --- EMBEDDED LUA CSV PARSER ENGINE ---

\begin{luacode*}
function parse_csv_line(line)
    local res = {}
    local pos = 1
    -- Ensure line ends cleanly with a separator to capture trailing empty cells
    line = line .. ',' 
    
    while pos <= #line do
        local next_char = string.sub(line, pos, pos)
        if next_char == '"' then
            -- Quoted field parsing
            local txt = ""
            pos = pos + 1
            while pos <= #line do
                local startp, endp = string.find(line, '^[^"]*', pos)
                txt = txt .. string.sub(line, pos, endp)
                pos = endp + 1
                if string.sub(line, pos, pos) == '"' then
                    if string.sub(line, pos + 1, pos + 1) == '"' then
                        txt = txt .. '"' -- Escaped quote sign inside cell
                        pos = pos + 2
                    else
                        pos = pos + 1 -- Outer closing quote sign reached
                        break
                    end
                end
            end
            table.insert(res, txt)
            pos = pos + 1 -- Skip the trailing comma delimiter
        else
            -- Plain unquoted field extraction
            local startp, endp = string.find(line, '^[^,]*', pos)
            if endp then
                table.insert(res, string.sub(line, pos, endp))
                pos = endp + 2
            else
                break
            end
        end
    end
    return res
end

-- Safely sanitize critical LaTeX control characters found inside strings
function escape_latex(text)
    if not text then return "" end
    text = text:gsub("\\", "\\textbackslash{}")
    text = text:gsub("([&%%%$#{}_])", "\\%1")
    text = text:gsub("~", "\\textasciitilde{}")
    text = text:gsub("%^", "\\textasciicircum{}")
    return text
end

-- Bold inline sense numbers in multi-definition entries. Matches
-- "\s\d{1,2}.\s" (e.g. " 2. ", " 12. ") so ordinary sentence-ending
-- digits like "series #2." followed by a parenthesis are left untouched.
-- Also anchors the leading sense at the very start of the field ("1. ...").
-- Apply AFTER escape_latex (digits, dots and spaces are never escaped).
function format_senses(text)
    text = text:gsub("^(%d%d?%.)(%s)", "\\textbf{%1}%2")
    text = text:gsub("(%s)(%d%d?%.)(%s)", "%1\\textbf{%2}%3")
    return text
end

-- Italicize parenthesized usage labels in definitions: regionalisms
-- "(AmEn, UK)", attitude "(derog.)", or combinations "(UK, slang, derog.)".
-- A label is recognized in expansion-initial position only, i.e. either at
-- the very start of a sole unnumbered definition, or right after a full
-- stop. The latter covers both a sense number ("2. (UK) ...", since "2."
-- itself ends in a stop) and a new expansion within a sense ("...; sour.
-- (of wine) high in acidity."). Parentheticals elsewhere in the running
-- text are left untouched. The parentheses stay upright; only the content
-- is italic. Nested parentheses are not supported. The two patterns cannot
-- overlap (field start has no preceding stop), so no double-wrapping.
-- Apply AFTER escape_latex but BEFORE format_senses, so this sees the raw
-- "1. " form, not the inserted \textbf{}.
function format_labels(text)
    -- Sole unnumbered definition: label at the very start of the field
    text = text:gsub("^(%s*)%(([^()]+)%)", "%1(\\textit{%2})")
    -- Label right after a full stop: sense number or expansion boundary
    text = text:gsub("(%.%s+)%(([^()]+)%)", "%1(\\textit{%2})")
    return text
end

-- Entries whose raw definition exceeds this many characters abandon the
-- keep-whole policy for the sense-level fallback: breaks allowed between
-- numbered senses (and between their examples), never inside one. Roughly
-- a third of a column; tune to taste.
SENSE_SPLIT_THRESHOLD = 600

-- Split a raw (unformatted) definition at sense-number boundaries, i.e.
-- the same "\s\d{1,2}.\s" shape format_senses embolden. Returns the list
-- of fragments: {"1. first sense ...", "2. second sense ...", ...}.
-- Run BEFORE escape_latex/format_* -- each fragment then flows through
-- the normal formatting pipeline, whose anchored patterns treat the
-- fragment start exactly like a field start ("2. (label) ..." works).
function split_senses(text)
    local parts = {}
    local last, init = 1, 1
    while true do
        local s = text:find("%s%d%d?%.%s", init)
        if not s then break end
        table.insert(parts, text:sub(last, s - 1))
        last = s + 1
        init = s + 1
    end
    table.insert(parts, text:sub(last))
    return parts
end

-- Parse the Examples column: a semicolon separated list of
-- "example expression: definition" pairs. Definitions may themselves
-- contain semicolons: an item WITHOUT a colon is not a new example but
-- the continuation of the previous definition, and is rejoined with ";".
-- ("compression ~: ... permanent deformation; the tendency of ..." stays
-- one example.) Limitation: a continuation fragment containing a colon
-- is indistinguishable from a new example.
-- For large entries (breakable=true) only the first example is glued to
-- the entry; the rest may break to the next column/page.
-- Returns a string of \printexample calls, or "" when the cell is empty.
function parse_examples(text, breakable)
    if not text or text:gsub("%s+", "") == "" then return "" end
    local items = {}
    for item in text:gmatch("[^;]+") do
        if item:find(":") then
            table.insert(items, item)
        elseif #items > 0 then
            items[#items] = items[#items] .. ";" .. item
        end
    end
    local out = {}
    for i, item in ipairs(items) do
        -- Split on the first colon only; definitions may contain further colons
        local expr, def = item:match("^%s*(.-)%s*:%s*(.-)%s*$")
        if expr and expr ~= "" then
            local cmd = (breakable and i > 1)
                        and "\\printexamplecont" or "\\printexample"
            table.insert(out, string.format("%s{%s}{%s}",
                cmd, escape_latex(expr), escape_latex(def)))
        end
    end
    return table.concat(out)
end

-- Case folding that also covers non-ASCII when LuaTeX's unicode lib exists
local ulower = (unicode and unicode.utf8 and unicode.utf8.lower) or string.lower

-- Load the custom collation alphabet: every sortable "letter" (possibly a
-- multi-character digraph like "Ll" or a diacritic like "Č") separated by
-- commas and/or blanks -- both work, mixed freely. Returns nil if absent.
function load_sort_order(filename)
    local f = io.open(filename, "r")
    if not f then return nil end
    local content = f:read("*a")
    f:close()
    local letters = {}
    for tok in content:gmatch("[^,%s]+") do
        table.insert(letters, tok)
    end
    if #letters == 0 then return nil end
    return letters
end

-- Build collator state: rank per lowercased letter, plus the letter list
-- ordered longest-first so digraphs ("ll") beat their prefix letter ("l")
function build_collator(letters)
    local rank, probe = {}, {}
    for i, l in ipairs(letters) do
        local ll = ulower(l)
        rank[ll] = i
        table.insert(probe, ll)
    end
    table.sort(probe, function(a, b) return #a > #b end)
    return rank, probe
end

-- Turn a word into its sequence of sort ranks via longest-match. Anything
-- not listed in the sort order sinks behind all real letters (1000 + byte),
-- so digits and stray symbols never claim a letter block of their own.
function sort_tokens(word, rank, probe)
    local w = ulower(word)
    local tokens = {}
    local pos = 1
    while pos <= #w do
        local matched = false
        for _, l in ipairs(probe) do
            if w:sub(pos, pos + #l - 1) == l then
                table.insert(tokens, rank[l])
                pos = pos + #l
                matched = true
                break
            end
        end
        if not matched then
            table.insert(tokens, 1000 + w:byte(pos))
            pos = pos + 1
        end
    end
    return tokens
end

-- Encode a token sequence as a fixed-width sort key for makeindex, used
-- via \index{key@word}. Each rank becomes three base-26 lowercase letters
-- ("aab", "aac", ...): makeindex would sort digit-leading keys numerically
-- (its "numbers" group), but pure-letter keys compare cleanly byte-wise,
-- making the index collation match ours exactly. Capacity 26^3 = 17576
-- comfortably covers alphabet ranks plus the 1000+byte unknown-char sink.
function index_key(tokens)
    local parts = {}
    for _, t in ipairs(tokens) do
        local n = t
        local c3 = n % 26; n = math.floor(n / 26)
        local c2 = n % 26; n = math.floor(n / 26)
        local c1 = n % 26
        table.insert(parts, string.char(97 + c1, 97 + c2, 97 + c3))
    end
    return table.concat(parts)
end

-- Main engine pipeline
function process_dictionary_csv(filename)
    local file = io.open(filename, "r")
    if not file then
        tex.print("\\textbf{Error: CSV file not found!}")
        return
    end

    -- Pass 1: collect all rows
    local entries = {}
    local is_header = true
    for line in file:lines() do
        if is_header then
            is_header = false
        else
            -- Strip trailing windows/unix carriage returns smoothly
            line = line:gsub("[\r\n]", "")
            if line:gsub("%s+", "") ~= "" then
                local data = parse_csv_line(line)
                if #data >= 4 then
                    table.insert(entries, data)
                end
            end
        end
    end
    file:close()

    -- Pass 2: optional custom collation (sortorder.txt). Without the file
    -- the entries flow through untouched, in CSV order, without headers.
    local letters = load_sort_order("sortorder.txt")
    if letters then
        local rank, probe = build_collator(letters)
        for _, e in ipairs(entries) do
            e.tokens = sort_tokens(e[1], rank, probe)
        end
        table.sort(entries, function(a, b)
            local ta, tb = a.tokens, b.tokens
            for i = 1, math.min(#ta, #tb) do
                if ta[i] ~= tb[i] then return ta[i] < tb[i] end
            end
            if #ta ~= #tb then return #ta < #tb end
            return a[1] < b[1] -- deterministic tie-break
        end)
    end

    -- Pass 3: ship entries, opening a letter block whenever the leading
    -- letter changes. The header shows the letter as written in sortorder.txt.
    local current = nil
    for _, e in ipairs(entries) do
        if letters then
            local first = e.tokens[1]
            if first and first <= #letters and first ~= current then
                current = first
                tex.print(string.format("\\letterheader{%s}", escape_latex(letters[first])))
            end
        end
        local word = escape_latex(e[1])
        local ipa  = e[2] -- Retain raw string safely for font structures
        local pos  = escape_latex(e[3])

        -- Log word and page number. With a custom sort order, prefix the
        -- makeindex "key@display" sort key so the index collates identically.
        if letters then
            tex.print(string.format("\\index{%s@%s}", index_key(e.tokens), word))
        else
            tex.print(string.format("\\index{%s}", word))
        end

        -- Ship parameters directly back into TeX macros. Small entries are
        -- one unbreakable block; large ones fall back to sense-level breaks.
        local pieces = {}
        local large = #e[4] > SENSE_SPLIT_THRESHOLD
        if large then
            local senses = split_senses(e[4])
            local head = table.remove(senses, 1)
            pieces[#pieces + 1] = string.format("\\printentry{%s}{%s}{%s}{%s}",
                word, ipa, pos, format_senses(format_labels(escape_latex(head))))
            for _, s in ipairs(senses) do
                pieces[#pieces + 1] = string.format("\\printsense{%s}",
                    format_senses(format_labels(escape_latex(s))))
            end
        else
            local definition = format_senses(format_labels(escape_latex(e[4])))
            pieces[#pieces + 1] = string.format("\\printentry{%s}{%s}{%s}{%s}",
                word, ipa, pos, definition)
        end

        -- Optional 5th column: examples, one \printexample per item
        local examples = parse_examples(e[5], large)
        if examples ~= "" then
            pieces[#pieces + 1] = examples
        end

        -- Close the entry: \entrydone (clears the continuation mark) must
        -- sit INSIDE the entry's final paragraph so the mark migrates out
        -- attached to its last typeset line. Emitted as a separate vertical
        -- mode token it may be separated from the entry by a page break,
        -- leaving a stale "open" mark on the wrong page. Every emitted piece
        -- ends with the closing brace of a macro argument, so we inject it
        -- just before that brace.
        pieces[#pieces] = pieces[#pieces]:sub(1, -2) .. "\\entrydone}"
        for _, p in ipairs(pieces) do
            tex.print(p)
        end
    end
end
