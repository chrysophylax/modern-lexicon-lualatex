% Compile with LuaLaTeX only -- the dictionary body is generated at
% compile time by the Lua engine in dictionary-tools.lua, loaded through
% the dictionary-tools package. This file holds design and content only.
\documentclass[10pt,twoside,twocolumn]{article}
\usepackage{geometry}
\geometry{paperwidth=180mm, paperheight=250mm, inner=22mm, outer=18mm, top=25mm, bottom=22mm, headsep=6mm, footskip=10mm}

\usepackage{fontspec}
\usepackage{microtype}

% fallback table to Dejavu Serif if we are missing a symbol
\directlua{
  luaotfload.add_fallback (
    "scriptfallback",
    {
      "DejaVuSerif:mode=node;script=latn;"
    }
  )
}


% Fonts are vendored in fonts/ (see ACKNOWLEDGEMENTS.md for licenses).
% Gentium covers Latin, Greek and Cyrillic -- our headword scripts.
\setmainfont{Gentium}[
  Path=./fonts/, Extension=.ttf,
  UprightFont=*-Regular, BoldFont=*-Bold,
  ItalicFont=*-Italic, BoldItalicFont=*-BoldItalic,
  Ligatures = TeX,
  RawFeature = {fallback=scriptfallback}]

% Entry macros, running-head marks and the CSV engine
\usepackage{dictionary-tools}
% abbreviations table
\usepackage{dictionary-abbreviations}

% Charis SIL has the richest IPA coverage; use it for transcriptions only
\newfontfamily\charisipafont{CharisSIL}[
  Path=./fonts/, Extension=.ttf,
  UprightFont=*-Regular, BoldFont=*-Bold,
  ItalicFont=*-Italic, BoldItalicFont=*-BoldItalic]
\renewcommand{\dictipafont}{\charisipafont}

% --- GRID AND TYPOGRAPHY LOCKS ---
% Entries are set at 9.5pt/14pt (\dictfontsize from dictionary-tools);
% \flushbottom locks every page to that 14pt baseline grid.
\renewcommand{\baselinestretch}{1.0}
\setlength{\emergencystretch}{2em}
\setlength{\parskip}{0pt}
\flushbottom

% --- index ---
\makeindex[options=-q]

% --- RUNNING HEADERS ---
% Layout is the document's business; the dictword/dictcont marks and
% \contnote come from dictionary-tools.
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\fancyhead[LE,RO]{\small\bfseries\thepage}
\fancyhead[LO,RE]{\small\bfseries\FirstMark[page]{dictword}\ \ --\ \ \LastMark[page]{dictword}\contnote}
\renewcommand{\headrulewidth}{0.4pt}

% here our actual document begins
% let there be book
\begin{document}
\clearpage
\onecolumn
\pagenumbering{roman}

% the pages before the actual dictionary
\include{cover}
\include{copyright}
\include{thanks}
\include{foreword}
% print our abbreviations page
%\thispagestyle{empty}
\printabbreviations{abbreviations.csv}
\include{notes}
\include{forepage}
% build the lexicon from our csv
\twocolumn
\pagenumbering{arabic}
\builddictionary{lexicon.csv}
\clearpage
\raggedbottom % allow undedfull vbox for last entries
% 3. Switch back to a cleaner header format for the index section
\fancyhead[LO,RE]{\small\bfseries INDEX}
\printindex
\end{document}
